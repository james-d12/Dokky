#pragma once 
#ifndef ATOM_MATH_LINEAR_ALGEBRA_DYNAMIC_VECTOR_ENGINE_HPP
#define ATOM_MATH_LINEAR_ALGEBRA_DYNAMIC_VECTOR_ENGINE_HPP

#include <core/utils.hpp>
#include <vector>

namespace atm::math {

template<typename T>
class dynamic_vector_engine
{
	static_assert(is_scalar_element_v<T>, "dynamic_vector_engine template parameter must be a scalar type!");
public:
	using type = T;
	using pointer = type*;
	using reference = type&;
	using const_reference = const type&;
	using size_type = std::size_t;
	using storage_type = std::vector<type>
	using engine_type = resizable_vector_engine_type;

	using iterator = typename storage_type::iterator;
	using const_iterator = typename storage_type::const_iterator;
	using reverse_iterator = typename storage_type::reverse_iterator;
	using const_reverse_iterator = typename storage_type::const_reverse_iterator;

	dynamic_vector_engine() = default;
	dynamic_vector_engine(const dynamic_vector_engine&) = default;
	dynamic_vector_engine& operator=(const dynamic_vector_engine&) = default;
	dynamic_vector_engine(dynamic_vector_engine&&) noexcept = default;
	dynamic_vector_engine& operator=(dynamic_vector_engine&&) noexcept = default;
	~dynamic_vector_engine() = default;

	constructs a dynamic_vector_engine with a given starting size, and sets each element to a given value.
	explicit dynamic_vector_engine(size_type num, const_reference val = type());
	constructs a dynamic_vector_engine with a storage_type data structure.
	explicit dynamic_vector_engine(const storage_type data);
	constructs a dynamic_vector_engine with an std::initializer_list data structure.
	dynamic_vector_engine(const std::initializer_list<type> data);
	copies a storage_type data structure into the dynamic_vector_engine.
	dynamic_vector_engine& operator=(const storage_type data);
	copies an std::initializer_list data structure into the dynamic_vector_engine.
	dynamic_vector_engine& operator=(const std::initializer_list<type> data);

	returns a reference to an element at the given index.
	ATOM_NODISCARD reference operator[](const size_type index) noexcept;
	returns a const reference to an element at the given index.
	ATOM_NODISCARD const_reference operator[](const size_type index) const noexcept;

	return the begin iterator.
	auto begin() noexcept;
	return the begin iterator without modifying elements.
	auto begin() const noexcept;
	return the end iterator.
	auto end() noexcept;
	return the end iterator without modifying elements.
	auto end() const noexcept;

	returns the reverse begin iterator.
	auto rbegin() noexcept;
	returns the reverse begin iterator without modifying elements.
	auto rbegin() const noexcept;
	returns the reverse end iterator.
	auto rend() noexcept;
	returns the reverse end iterator without modifying elements.
	auto rend() const noexcept;

	returns the const begin iterator.
	auto cbegin() const noexcept;
	returns the const end iterator.
	auto cend() const noexcept;
	returns the const reverse begin iterator.
	auto crbegin() const noexcept;
	returns the const reverse end iterator.
	auto crend() const noexcept;

	returns the underlying data as an instance of the storage_type.
	ATOM_NODISCARD auto data() const noexcept;
	returns the total number of elements within the fixed_vector_engine.
	ATOM_NODISCARD auto size() const noexcept;
	returns the total number of elements possible within the fixed_vector_engine.
	ATOM_NODISCARD auto capacity() const noexcept;

	adjusts the current size of the dynamic_vector_engine to the given size.
	void resize(size_type size);
	adjusts the maximum capacity of the dynamic_vector_engine to the given capacity.
	void reserve(size_type capacity);
	adjusts both the size and capacity of the dynamic_vector_engine in one.
	void resize_and_reserve(size_type size, size_type capacity);

	sets each element within the fixed_vector_engine to a given value.
	void fill(const_reference n);

private:
	storage_type m_data;
};

template<typename T>
dynamic_vector_engine<T>::dynamic_vector_engine(const size_type num, const_reference val)
: m_data(num, val)
{

}

template<typename T>
dynamic_vector_engine<T>::dynamic_vector_engine(const storage_type data)
: m_data(data)
{

}

template<typename T>
dynamic_vector_engine<T>::dynamic_vector_engine(const std::initializer_list<type> data)
{
	std::fill(begin(), end(), T(0));
	resize(data.size());
	std::copy(std::cbegin(data), std::cend(data), m_data.begin());
}

template<typename T>
dynamic_vector_engine<T>&
dynamic_vector_engine<T>::operator=(const storage_type data)
{
	m_data = data;
	return *this;
}

template<typename T>
dynamic_vector_engine<T>&
dynamic_vector_engine<T>::operator=(const std::initializer_list<type> data)
{
	resize(data.size());
	std::copy(std::cbegin(data), std::cend(data), m_data.begin());
}
