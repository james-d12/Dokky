#pragma once 
#ifndef ATOM_MATH_LINEAR_ALGEBRA_FIXED_VECTOR_ENGINE_HPP
#define ATOM_MATH_LINEAR_ALGEBRA_FIXED_VECTOR_ENGINE_HPP

#include "linear_algebra_traits.hpp"
#include <array>

namespace atm::math {

template<typename T, std::size_t D>
class fixed_vector_engine
{
	static_assert(is_scalar_element_v<T>, "fixed_vector_engine template parameter must be a scalar type!");
	static_assert(D > 0, "Dimension size must be greater than 0!");

/* Problematic at the moment
#if defined ATOM_ENABLE_DEBUG_MODE
	static_assert(D > 10000, "Fixed_vector_engine should not be used to store more than 10,000 elements, use dynamic_vector_engine instead!");
#endif 
*/

public:
	using type = T;
	using pointer = type*;
	using reference = type&;
	using const_reference = const type&;
	using size_type = std::size_t;
	using storage_type = std::array<type, D>
	using engine_type = fixed_vector_engine_type;

	using iterator = typename storage_type::iterator;
	using const_iterator = typename storage_type::const_iterator;
	using reverse_iterator = typename storage_type::reverse_iterator;
	using const_reverse_iterator = typename storage_type::const_reverse_iterator;

	constexpr fixed_vector_engine(const fixed_vector_engine&) = default;
	constexpr fixed_vector_engine& operator=(const fixed_vector_engine&) = default;
	constexpr fixed_vector_engine(fixed_vector_engine&&) noexcept = default;
	constexpr fixed_vector_engine& operator=(fixed_vector_engine&&) noexcept = default;
	~fixed_vector_engine() noexcept = default;

	constructs an empty fixed_vector_engine.
	constexpr fixed_vector_engine();
	constructs a fixed_vector_engine with a storage_type data structure.
	constexpr explicit fixed_vector_engine(const storage_type data);
	constructs a fixed_vector_engine with an std::initializer_list data structure.
	constexpr fixed_vector_engine(const std::initializer_list<type> data);
	copies a storage_type data structure into the fixed_vector_engine.
	constexpr fixed_vector_engine& operator=(const storage_type data);
	copies an std::initializer_list data structure into the fixed_vector_engine.
	constexpr fixed_vector_engine& operator=(const std::initializer_list<type> data);

	returns a reference to an element at the given index.
	ATOM_NODISCARD constexpr reference operator[](const size_type index) noexcept;
	returns a const reference to an element at the given index.
	ATOM_NODISCARD constexpr const_reference operator[](const size_type index) const noexcept;

	return the begin iterator.
	constexpr auto begin() noexcept;
	return the begin iterator without modifying elements.
	constexpr auto begin() const noexcept;
	return the end iterator.
	constexpr auto end() noexcept;
	return the end iterator without modifying elements.
	constexpr auto end() const noexcept;

	returns the reverse begin iterator.
	constexpr auto rbegin() noexcept;
	returns the reverse begin iterator without modifying elements.
	constexpr auto rbegin() const noexcept;
	returns the reverse end iterator.
	constexpr auto rend() noexcept;
	returns the reverse end iterator without modifying elements.
	constexpr auto rend() const noexcept;

	returns the const begin iterator.
	constexpr auto cbegin() const noexcept;
	returns the const end iterator.
	constexpr auto cend() const noexcept;
	returns the const reverse begin iterator.
	constexpr auto crbegin() const noexcept;
	returns the const reverse end iterator.
	constexpr auto crend() const noexcept;

	returns the underlying data as an instance of the storage_type.
	ATOM_NODISCARD constexpr auto data() const noexcept;
	returns the total number of elements within the fixed_vector_engine.
	ATOM_NODISCARD constexpr auto size() const noexcept;
	returns the total number of elements possible within the fixed_vector_engine.
	ATOM_NODISCARD constexpr auto capacity() const noexcept;

	sets each element within the fixed_vector_engine to a given value.
	constexpr void fill(const_reference n);

private:
	storage_type m_data;
};

template<typename T, std::size_t D>
constexpr fixed_vector_engine<T, D>::fixed_vector_engine()
{
	m_data.fill(0);
}

template<typename T, std::size_t D>
constexpr fixed_vector_engine<T, D>::fixed_vector_engine(const storage_type data)
	: m_data(data)
{
}

template<typename T, std::size_t D>
constexpr fixed_vector_engine<T, D>::fixed_vector_engine(const std::initializer_list<type> data)
{
	std::copy(std::begin(data), std::end(data), m_data.begin());
}

template<typename T, std::size_t D>
constexpr fixed_vector_engine<T, D>&
fixed_vector_engine<T, D>::operator=(const storage_type data)
{
	m_data = data;
	return *this;
}

template<typename T, std::size_t D>
constexpr fixed_vector_engine<T, D>&
fixed_vector_engine<T, D>::operator=(const std::initializer_list<type> data)
{
	std::copy(std::begin(data), std::end(data), m_data.begin());
	return *this;
}

template<typename T, std::size_t D>
constexpr typename fixed_vector_engine<T, D>::reference
fixed_vector_engine<T, D>::operator[](const size_type index) noexcept
{
	return m_data[index];
}

template<typename T, std::size_t D>
constexpr typename fixed_vector_engine<T, D>::const_reference
fixed_vector_engine<T, D>::operator[](const size_type index) const noexcept
{
	return m_data[index];
}
