template<class ENG>
class matrix
{
	static_assert(is_engine_matrix_v<ENG>, "Engine parameter must be a matrix engine.");
public:
	using type = typename ENG::type;
	using reference = typename ENG::reference;
	using const_reference = typename ENG::const_reference;
	using size_type = typename ENG::size_type;
	using storage_type = typename ENG::storage_type;
	using engine_type = typename ENG::engine_type;

	using iterator = typename ENG::iterator;
	using const_iterator = typename ENG::const_iterator;
	using reverse_iterator = typename ENG::reverse_iterator;
	using const_reverse_iterator = typename ENG::const_reverse_iterator;

	constexpr matrix(const matrix&) = default;
	constexpr matrix& operator=(const matrix&) = default;
	constexpr matrix(matrix&&) noexcept = default;
	constexpr matrix& operator=(matrix&&) noexcept = default;
	~matrix() = default;

	constructs an empty matrix.
	constexpr matrix();
	constructs a matrix with a storage_type data structure.
	constexpr explicit matrix(const storage_type data);
	constructs a matrix with an std::initializer_list data structure.
	constexpr matrix(const std::initializer_list<type> data);
	constructs a matrix with an engine provided.
	constexpr explicit matrix(const ENG& engine);
	copies a storage_type data structure into the matrix.
	constexpr matrix& operator=(const storage_type data);
	copies an std::initializer_list data structure into the matrix.
	constexpr matrix& operator=(const std::initializer_list<type> data);
	copies an engine into the matrix.
	constexpr matrix& operator=(const ENG& engine);

	returns a reference to an element at a given index.
	ATOM_NODISCARD constexpr reference operator[](const size_type index);
	returns a const reference to an element at a given index.
	ATOM_NODISCARD constexpr const_reference operator[](const size_type index) const;

	returns a reference to an element at the given i and j indexes.
	ATOM_NODISCARD constexpr reference operator()(const size_type i, const size_type j);
	returns a const reference to an element at the given i and j indexes.
	ATOM_NODISCARD constexpr const_reference operator()(const size_type i, const size_type j) const;

	returns the order of the matrix (e.g. row / column major.)
	ATOM_NODISCARD constexpr auto order() const noexcept;
	returns an instance of the underlying engine type.
	ATOM_NODISCARD constexpr auto engine() const noexcept;
	returns the underlying data as an instance of the storage_type.
	ATOM_NODISCARD constexpr auto data() const noexcept;
	returns the total number of elements within the matrix.
	ATOM_NODISCARD constexpr auto dimensions() const noexcept;
	returns the total number of elements possible within the matrix.
	ATOM_NODISCARD constexpr auto capacity() const noexcept;
	returns the total number of rows.
	ATOM_NODISCARD constexpr auto rows() const noexcept;
	returns the total number of rows possible.
	ATOM_NODISCARD constexpr auto rows_capacity() const noexcept;
	returns the total number of columns.
	ATOM_NODISCARD constexpr auto columns() const noexcept;
	returns the total number of columns possible.
	ATOM_NODISCARD constexpr auto columns_capacity() const noexcept;

	returns an entire row.
	ATOM_NODISCARD constexpr auto get_row(size_type index) const;
	returns an entire column.
	ATOM_NODISCARD constexpr auto get_column(size_type index) const;

	sets each element within the matrix to a given value.
	constexpr void fill(const_reference n);

	outputs every element within the matrix.
	void print() const noexcept;

	transposes the matrix.
	void tranpose();
	returns a tranpose of the matrix.
	ATOM_NODISCARD constexpr auto transposed() const;

private:
	ENG m_engine;
};

template<class ENG>
void matrix<ENG>::print() const noexcept
{
	size_type index = 0;
	switch(order())
	{
		case MatrixOrder::Row:
			for (size_type i = 0; i < rows(); i++)
			{
				std::cout << "[ ";
				for (size_type j = 0; j < columns(); j++)
				{
					index = j + (i * rows());
					std::cout << m_engine[index] << " ";
				}
				std::cout << "]\n";
			}
			break;
		case MatrixOrder::Column:
			for (size_type i = 0; i < rows(); i++)
			{
				std::cout << "[ ";
				for (size_type j = 0; j < columns(); j++)
				{
					index = i + (j * rows());
					std::cout << m_engine[index] << " ";
				}
				std::cout << "]\n";
			}
			break;
		default:
			break;
	}
}

template<class ENG1, class ENG2>
adds together 2 matrices of varying engine types.
inline auto operator+(const matrix<ENG1>& lhs, const matrix<ENG2>& rhs)
{
	atm::assertion(lhs.dimensions() == rhs.dimensions(), "Matrices must be equal sizes to perform addition.");
	auto result_engine = detail::matrix_engine_addition(lhs.engine(),rhs.engine());
	using result_engine_type = decltype(result_engine);
	return matrix<result_engine_type>(result_engine);
}

template<class ENG1, class ENG2>
subtracts 2 matrices of varying engine types.
inline auto operator-(const matrix<ENG1>& lhs, const matrix<ENG2>& rhs)
{
	atm::assertion(lhs.dimensions() == rhs.dimensions(), "Matrices must be equal sizes to perform subtraction.");
	auto result_engine = detail::matrix_engine_subtraction(lhs.engine(),rhs.engine());
	using result_engine_type = decltype(result_engine);
	return matrix<result_engine_type>(result_engine);
}
